297 static int safe_vfscanf(FILE *stream, const char *format, va_list ap)
298 {
299  int r;
300  do
301  {
302  if (ferror(stream) && errno == EINTR)
303  {
304  clearerr(stream);
305  }
306  r = ::vfscanf(stream, format, ap);
307  }
308  while (r == EOF && ferror(stream) && errno == EINTR);
309  return r;
310 }
