70 InetAddr::InetAddr(const sockaddr* a) : InetAddr()
71 {
72  if (a->sa_family == AF_INET)
73  {
74  const sockaddr_in* x = reinterpret_cast<const sockaddr_in*>(a);
75  // convert to mapped IPv4 address
76  //uint32_t h = ntohl(x->sin_addr.s_addr);
77  uint32_t h = x->sin_addr.s_addr;
78  unsigned char* b = reinterpret_cast<unsigned char*>(&h);
79  m_addr->sin6_addr.s6_addr[10] = 0xff;
80  m_addr->sin6_addr.s6_addr[11] = 0xff;
81  m_addr->sin6_addr.s6_addr[12] = b[0];
82  m_addr->sin6_addr.s6_addr[13] = b[1];
83  m_addr->sin6_addr.s6_addr[14] = b[2];
84  m_addr->sin6_addr.s6_addr[15] = b[3];
85 
86  m_addr->sin6_port = x->sin_port;
87  }
88  else if (a->sa_family == AF_INET6)
89  {
90  memcpy(m_addr, a, sizeof(sockaddr_in6));
91  }
92 }

210 int InetAddr::flags() const
211 {
212  int flags = 0;
213 
214  const unsigned char* ad = m_addr->sin6_addr.s6_addr;
215  if (btest(ad, 0, 10) && ad[10] == 0xff && ad[11] == 0xff)
216  {
217  flags |= InetAddrFlag::ipv4;
218 
219  if (btest(ad+12, 0, 4))
220  {
221  flags |= InetAddrFlag::any;
222  }
223  else if (ad[12] == 127)
224  {
225  flags |= InetAddrFlag::loopback;
226  }
227  else if (ad[12] >= 224 && ad[12] <= 239)
228  {
229  flags |= InetAddrFlag::multicast;
230  flags |= InetAddrFlag::global;
231  }
232  else
233  {
234  flags |= InetAddrFlag::unicast;
235  flags |= InetAddrFlag::global;
236  }
237  }
238  else
239  {
240  flags |= InetAddrFlag::ipv6;
241 
242  if (btest(ad, 0, 16)) // ::
243  {
244  flags |= InetAddrFlag::any;
245  }
246  else if (btest(ad, 0, 15) && ad[15] == 1) // ::1
247  {
248  flags |= InetAddrFlag::loopback;
249  }
250  else if (ad[0] == 0xff)
251  {
252  flags |= InetAddrFlag::multicast;
253  // flags
254  if ((ad[1] & 0x10) == 0x10)
255  {
256  flags |= InetAddrFlag::mcast_dynamic;
257  }
258  if ((ad[1] & 0x20) == 0x10)
259  {
260  flags |= InetAddrFlag::mcast_prefix;
261  }
262  if ((ad[1] & 0x40) == 0x10)
263  {
264  flags |= InetAddrFlag::mcast_rendezvous;
265  }
266  // scope
267  switch (ad[1] & 0xf)
268  {
269  case 0x1: flags |= InetAddrFlag::if_local; break;
270  case 0x2: flags |= InetAddrFlag::link_local; break;
271  case 0x3: flags |= InetAddrFlag::realm_local; break;
272  case 0x4: flags |= InetAddrFlag::admin_local; break;
273  case 0x5: flags |= InetAddrFlag::site_local; break;
274  case 0x8: flags |= InetAddrFlag::org_local; break;
275  case 0xe: flags |= InetAddrFlag::global; break;
276  }
277  // reserved addresses ff00::/24
278  if ((ad[1] & 0xf0) == 0x00)
279  {
280  if (btest(ad+2, 0, 13) && ad[15] == 1) // ff0X::1
281  {
282  flags |= InetAddrFlag::mcast_all_nodes;
283  }
284  if (btest(ad+2, 0, 13) && ad[15] == 2) // ff0X::2
285  {
286  flags |= InetAddrFlag::mcast_all_routers;
287  }
288  }
289  }
290  else
291  {
292  flags |= InetAddrFlag::unicast;
293 
294  // fe80::/64
295  if (ad[0] == 0xfe && ad[1] == 0x80 && btest(ad+2, 0, 6)) // 8 bytes total
296  {
297  flags |= InetAddrFlag::link_local;
298  }
299  else
300  {
301  flags |= InetAddrFlag::global;
302  }
303 
304  // fc00::/7; currently fd00::/8 is defined for /48 prefixes
305  if ((ad[0] & 0xfe) == 0xfc)
306  {
307  flags |= InetAddrFlag::unique_local_address;
308  }
309  }
310  }
311 
312  return flags;
313 }

320 std::string InetAddr::str() const
321 {
322  std::stringstream s;
323  int f = flags();
324  switch (f & InetAddrFlag::prot_mask)
325  {
326  case InetAddrFlag::ipv4:
327  s << "ipv4";
328  break;
329  case InetAddrFlag::ipv6:
330  s << "ipv6";
331  break;
332  }
333  s << " " << host();
334  s << " port: " << port() << " scope_id: " << scope_id();
335  s << " flags:";
336  switch (f & InetAddrFlag::type_mask)
337  {
338  case InetAddrFlag::any:
339  s << " type-any";
340  break;
341  case InetAddrFlag::loopback:
342  s << " type-loop";
343  break;
344  case InetAddrFlag::multicast:
345  s << " type-mcast";
346  break;
347  case InetAddrFlag::unicast:
348  s << " type-unicast";
349  break;
350  }
351 
352  switch (f & scope_mask)
353  {
354  case InetAddrFlag::if_local:
355  s << " scope-iface-local";
356  break;
357  case InetAddrFlag::link_local:
358  s << " scope-link-local";
359  break;
360  case InetAddrFlag::realm_local:
361  s << " scope-realm-local";
362  break;
363  case InetAddrFlag::admin_local:
364  s << " scope-admin-local";
365  break;
366  case InetAddrFlag::site_local:
367  s << " scope-site-local";
368  break;
369  case InetAddrFlag::org_local:
370  s << " scope-org-local";
371  break;
372  case InetAddrFlag::global:
373  s << " scope-global";
374  break;
375  }
376 
377  switch (f & InetAddrFlag::mcast_flags_mask)
378  {
379  case InetAddrFlag::mcast_rendezvous:
380  s << " mcast-flags-rendezvous";
381  break;
382  case InetAddrFlag::mcast_prefix:
383  s << " mcast-flags-prefix";
384  break;
385  case InetAddrFlag::mcast_dynamic:
386  s << " mcast-flags-dynamic";
387  break;
388  }
389 
390  switch (f & InetAddrFlag::mcast_reserved_mask)
391  {
392  case InetAddrFlag::mcast_all_nodes:
393  s << " mcast-all-nodes";
394  break;
395  case InetAddrFlag::mcast_all_routers:
396  s << " mcast-all-routers";
397  break;
398  }
399 
400  switch (f & InetAddrFlag::unicast_special_mask)
401  {
402  case InetAddrFlag::unique_local_address:
403  s << " unique-local-address";
404  break;
405  }
406 
407  return s.str();
408 }
