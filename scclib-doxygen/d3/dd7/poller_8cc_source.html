49 void Poller::set(int fd, int flags)
50 {
51  epoll_event ev;
52  int eflags = 0;
53 
54  if (flags & Poller::input)
55  {
56  eflags |= EPOLLIN;
57  }
58  if (flags & Poller::output)
59  {
60  eflags |= EPOLLOUT;
61  }
62  if (flags & Poller::read_hup)
63  {
64  eflags |= EPOLLRDHUP;
65  }
66  if (flags & Poller::priority)
67  {
68  eflags |= EPOLLPRI;
69  }
70 
71  if (eflags == 0)
72  {
73  throw std::runtime_error("No poll events specified");
74  }
75 
76  ev.events = eflags;
77  ev.data.fd = fd;
78 
79  if (m_polls.find(fd) == m_polls.end())
80  {
81  if (epoll_ctl(m_fd, EPOLL_CTL_ADD, fd, &ev) == -1)
82  {
83  throw std::system_error(errno, std::system_category());
84  }
85  }
86  else
87  {
88  if (epoll_ctl(m_fd, EPOLL_CTL_MOD, fd, &ev) == -1)
89  {
90  throw std::system_error(errno, std::system_category());
91  }
92  }
93  m_polls[fd] = eflags;
94 }

108 void Poller::wait(int timeout_ms)
109 {
110  epoll_event evs[m_polls.size()];
111  int n;
112  do
113  {
114  n = epoll_wait(m_fd, evs, m_polls.size(), timeout_ms);
115  }
116  while (n == -1 && errno == EINTR);
117  if (n == -1)
118  {
119  std::stringstream st;
120  st << "epoll_wait()";
121  throw std::system_error(errno, std::system_category(), st.str());
122  }
123  m_events.clear();
124  for (int i = 0; i < n; i++)
125  {
126  if (m_polls.find(evs[i].data.fd) == m_polls.end())
127  {
128  throw std::runtime_error("invalid poll event returned");
129  }
130  m_events[evs[i].data.fd] = converteflags(evs[i].events);
131  }
132 }
