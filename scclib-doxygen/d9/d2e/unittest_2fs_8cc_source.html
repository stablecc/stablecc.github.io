232 TEST(FsExampleTest, sparse_and_trunc)
233 {
234  system_error err;
235  fs::remove_all("sandbox", &err);
236  fs::create_dir("sandbox");
237  auto curdir = fs::get_current_dir();
238  cout << "curdir: " << curdir << endl;

240  fs::change_dir("sandbox");

242  string s("this is a test of the emergency sizing system\n");
243  cout << "sparse test with string len=" << s.size() << endl;
244 
245  string fn("sparse_file");
246 
247  auto write_sparse = [&fn, &s](off_t loc)
248  {
249  // use clib to write data into the middle of the file
250  // if we use fstream, the system may truncate the file
251 
252  // use system wrapper FileDesc to ensure the file is always closed
253  scc::util::FileDesc fd(scc::util::safe_open_throw(fn.c_str(), O_WRONLY));
254 
255  lseek(fd, loc, SEEK_SET);
256  safe_write_throw(fd, &s[0], s.size());
257 
258  cout << "wrote " << s.size() << " bytes at loc " << loc << endl;
259  };
260 
261  fs::create_reg(fn);
262  fs::set_size(fn, 16384); // 16 K file, with a 0 at the end
263 
264  auto st = fs::file_stat(fn);
265  cout << "12288 hole at start of file:\n" << st << endl;
266 
267  ASSERT_EQ(st.size, 16384);
268  ASSERT_EQ(st.alloc_size, 4096); // the last block has data
269 
270  auto sm = fs::sparse_map(fn);
271  cout << setw(6) << "start" << setw(6) << "data" << endl;
272  for (auto& x : sm)
273  {
274  cout << setw(6) << x.first << setw(6) << x.second << endl;
275  }
276 
277  std::map<int64_t,int64_t> ver;
278 
279  ver[0] = 12287; // 12288 sized hole at 0 (beginning)
280  ASSERT_EQ(sm, ver);
281 
282  write_sparse(1024); // write the string at 1024, causing first block to be written
283 
284  st = fs::file_stat(fn);
285  cout << "8192 hole in middle of file\n" << st << endl;
286 
287  ASSERT_EQ(st.alloc_size, 8192);
288 
289  sm = fs::sparse_map(fn);
290  cout << setw(6) << "start" << setw(6) << "data" << endl;
291  for (auto& x : sm)
292  {
293  cout << setw(6) << x.first << setw(6) << x.second << endl;
294  }
295 
296  ver.clear();
297  ver[4096] = 12287; // 8192 sized hole at 4096 (middle)
298  ASSERT_EQ(sm, ver);
299 
300  fs::set_size(fn, 12288);
301 
302  st = fs::file_stat(fn);
303  cout << "8192 hole at end of file\n" << st << endl;
304 
305  ASSERT_EQ(st.alloc_size, 4096);
306 
307  sm = fs::sparse_map(fn);
308  cout << setw(6) << "start" << setw(6) << "data" << endl;
309  for (auto& x : sm)
310  {
311  cout << setw(6) << x.first << setw(6) << x.second << endl;
312  }
313 
314  ver.clear();
315  ver[4096] = 12287; // 8192 sized hole at 4096 (end)
316  ASSERT_EQ(sm, ver);
317 
318  ifstream f(fn);
319  f.seekg(1024);
320  string val;
321  val.resize(s.size(), '\x01');
322  f.read(&val[0], val.size());
323  ASSERT_EQ(val, s);
324 
325  fs::change_dir(curdir);
326  fs::remove_all("sandbox");
327 }
